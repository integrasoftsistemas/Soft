-- Update na coluna criada logradouro com o logradouro do endereco
UPDATE VIACEP_Sem_GroupBy
SET logradouro = SUBSTRING(endereco, 1, CHARINDEX(' ', endereco, 1) - 1)

-- Copiando tabela VIACEP_Sem_GroupBy para VIACEP
INSERT INTO VIACEP
(cep, endereco, complemento, bairro, cidade, uf, unidade, ibge, gia, logradouro)
SELECT cep, endereco, complemento, bairro, cidade, uf, unidade, ibge, gia, logradouro
FROM VIACEP_Sem_GroupBy
GROUP BY cep, endereco, complemento, bairro, cidade, uf, unidade, ibge, gia, logradouro

-- Update na coluna endereco retirando o conteudo da coluna logradouro
UPDATE VIACEP
  SET endereco = REPLACE ( endereco, logradouro + ' ' , '' )
  
-- Criando coluna idLogradouro na tabela VIACEP
ALTER TABLE VIACEP ADD idLogradouro INT NULL ;

SELECT Logradouro.logID AS ID_Do_Logradouro
FROM Logradouro RIGHT OUTER JOIN
VIACEP ON Logradouro.logNome = VIACEP.logradouro */
